def cityName(x):
    cityName = ""
    ln = list(x)
    
    j = 1
    for k in ln:
        if (j ==1):
            cityName+= k.upper()
            j=j+1
        else:
            cityName+= k.lower()
    return cityName
    
    
df['temp_from']=df['temp_from'].apply(lambda x : cityName(x))
